[router2.js]
"/" : Dashboard or Singin
"/singin": Signin
"/signup" : Signup
"/dashboard" : Dashboard
"/dashboard/:user" : Dashboard (Master)
"/master" : MasterDashboard
"/meter/:channel" : Meter
"/powerq/:channel" : PowerQ
"/diagnosis/:channel" : Diagnosis
"/event/:channel" : Event
"/report/:channel" : Report2
"/trend/:channel" : Trend
"/settings/:channel" : Setting

[store.js]
user : login username
userRole : login userRole 0:User / 1:Operator / 2:Admin
opMode : device / server (Master Status)
logined : logined status
langStr : select language by login
setup : initial setup status
applysetup : applying status for setup
sessioncheckd : session status, login status
MainDiagnosis : setup status for Main channel Diagnosis
SubDiagnosis : setup status for Sub channel Diagnosis
MainEnable : setup status for Main channel
SubEnable : setup status for Sub channel
assetType_main : assetType for main channel diagnosis
assetName_main : assetName for main channel diagnosis
assetType_sub : assetType for sub channel diagnosis
assetName_sub : assetName for sub channel diagnosis
setupFromFile : upload status for setup
